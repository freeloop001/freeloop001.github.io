## 1. Embedding 向量嵌入

### 什么是 Embedding？

Embedding 将文本转换为向量表示，让语义相似的文本在向量空间中相近。

```python
from sentence_transformers import SentenceTransformer

# 中文效果好的模型
model = SentenceTransformer('bge-base-zh-v1.5')

# 生成向量
texts = ["Python是一门编程语言", "JavaScript是网页脚本语言"]
embeddings = model.encode(texts)

print(f"向量维度: {embeddings.shape}")
print(f"向量示例: {embeddings[0][:5]}")
```

### 常用 Embedding 模型

| 模型 | 维度 | 特点 | 适用场景 |
|------|------|------|----------|
| text-embedding-ada-002 | 1536 | OpenAI 官方 | 通用 |
| bge-base-zh-v1.5 | 768 | 中文效果优秀 | 中文 |
| m3e-base | 768 | 多语言支持 | 多语言 |
| bge-large-zh-v1.5 | 1024 | 效果最好 | 精度要求高 |

### Chunk 策略

```python
# 1. 固定大小分块
def fixed_chunk(text: str, chunk_size: int = 500, overlap: int = 50):
    chunks = []
    for i in range(0, len(text), chunk_size - overlap):
        chunks.append(text[i:i + chunk_size])
    return chunks

# 2. 递归分块
import re

def recursive_chunk(text: str, delimiters=["\n\n", "\n", ". ", " "]):
    if not delimiters:
        return [text]

    delimiter = delimiters[0]
    parts = text.split(delimiter)

    if len(parts) > 1:
        result = []
        for part in parts:
            result.extend(recursive_chunk(part, delimiters[1:]))
        return result
    else:
        return recursive_chunk(text, delimiters[1:])

# 3. 按段落分块
def paragraph_chunk(text: str):
    paragraphs = text.split("\n\n")
    return [p.strip() for p in paragraphs if p.strip()]
```

## 2. 向量数据库

### Chroma（轻量级）

```python
import chromadb
from sentence_transformers import SentenceTransformer

# 初始化
client = chromadb.Client()
collection = client.create_collection("docs")

# 加载模型
model = SentenceTransformer('bge-base-zh-v1.5')

# 文档
documents = [
    "Python是一门易学的编程语言",
    "JavaScript主要用于网页开发",
    "Rust是一门系统编程语言"
]

# 生成向量
embeddings = model.encode(documents)

# 添加文档
collection.add(
    documents=documents,
    embeddings=embeddings.tolist(),
    ids=["doc1", "doc2", "doc3"]
)

# 检索
query = "什么是Python？"
query_embedding = model.encode([query])

results = collection.query(
    query_embeddings=query_embedding.tolist(),
    n_results=2
)

print(results['documents'])
```

### Qdrant（云原生）

```python
from qdrant_client import QdrantClient
from qdrant_client.models import Distance, VectorParams
from sentence_transformers import SentenceTransformer

# 连接 Qdrant
client = QdrantClient(host="localhost", port=6333)

# 创建集合
client.create_collection(
    collection_name="docs",
    vectors_config=VectorParams(size=768, distance=Distance.COSINE)
)

# 加载文档
model = SentenceTransformer('bge-base-zh-v1.5')
documents = ["文档内容1", "文档内容2"]
embeddings = model.encode(documents)

# 添加向量
client.upsert(
    collection_name="docs",
    points=[
        {
            "id": 1,
            "vector": embeddings[0].tolist(),
            "payload": {"text": documents[0]}
        },
        {
            "id": 2,
            "vector": embeddings[1].tolist(),
            "payload": {"text": documents[1]}
        }
    ]
)

# 检索
query = "用户问题"
query_vector = model.encode([query])[0].tolist()

results = client.search(
    collection_name="docs",
    query_vector=query_vector,
    limit=3
)
```

### 向量数据库对比

| 数据库 | 类型 | 特点 | 适用场景 |
|--------|------|------|----------|
| Chroma | 轻量级 | 简单易用 | 原型开发 |
| Qdrant | 云原生 | 功能丰富 | 生产环境 |
| Milvus | 企业级 | 大规模、高并发 | 大规模部署 |
| FAISS | 本地 | 速度快 | 小规模 |

## 3. RAG 框架

### LangChain

```python
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.document_loaders import TextLoader
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings
from langchain_openai import ChatOpenAI
from langchain.chains import RetrievalQA

# 1. 加载文档
loader = TextLoader("document.txt")
documents = loader.load()

# 2. 分割文档
text_splitter = RecursiveCharacterTextSplitter(chunk_size=500, chunk_overlap=50)
docs = text_splitter.split_documents(documents)

# 3. 创建向量存储
embeddings = OpenAIEmbeddings()
vectorstore = Chroma.from_documents(docs, embeddings)

# 4. 创建 QA 链
qa = RetrievalQA.from_chain_type(
    llm=ChatOpenAI(),
    chain_type="stuff",
    retriever=vectorstore.as_retriever()
)

# 5. 问答
result = qa.invoke({"query": "文档主要内容是什么？"})
print(result["result"])
```

### LlamaIndex

```python
from llama_index.core import Document, VectorStoreIndex
from llama_index.llms.openai import OpenAI
from llama_index.embeddings.openai import OpenAIEmbedding

# 1. 创建文档
documents = [
    Document(text="Python是一门易学的编程语言"),
    Document(text="JavaScript主要用于前端开发")
]

# 2. 创建索引
index = VectorStoreIndex.from_documents(
    documents,
    embed_model=OpenAIEmbedding()
)

# 3. 创建查询引擎
query_engine = index.as_query_engine(llm=OpenAI(model="gpt-4o"))

# 4. 查询
response = query_engine.query("什么是Python？")
print(response)
```

### LangChain vs LlamaIndex

| 特性 | LangChain | LlamaIndex |
|------|-----------|------------|
| 定位 | LLM 应用框架 | 数据索引专家 |
| 灵活性 | 高 | 中 |
| 学习曲线 | 中等 | 较低 |
| 文档质量 | 中等 | 优秀 |
| 社区 | 非常活跃 | 活跃 |

### 使用 MCP 工具

```python
# MCP 服务器连接示例
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client

# 配置服务器
server_params = StdioServerParameters(
    command="npx",
    args=["-y", "@modelcontextprotocol/server-filesystem", "/tmp"]
)

# 连接并使用
async with stdio_client(server_params) as (read, write):
    async with ClientSession(read, write) as session:
        await session.initialize()

        # 调用工具
        result = await session.call_tool("read_file", {
            "path": "/tmp/example.txt"
        })
```

## 5. 实战项目

### 项目：本地知识库问答

```python
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.document_loaders import TextLoader
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain.chains import RetrievalQA

class LocalKnowledgeBase:
    def __init__(self, api_key: str, docs_path: str):
        self.llm = ChatOpenAI(api_key=api_key)
        self.embeddings = OpenAIEmbeddings(api_key=api_key)
        self.docs_path = docs_path

    def build_index(self):
        """构建知识库索引"""
        loader = TextLoader(self.docs_path)
        documents = loader.load()

        text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=500,
            chunk_overlap=50
        )
        docs = text_splitter.split_documents(documents)

        self.vectorstore = Chroma.from_documents(
            docs,
            self.embeddings,
            persist_directory="./chroma_db"
        )

    def query(self, question: str) -> str:
        """问答"""
        qa = RetrievalQA.from_chain_type(
            llm=self.llm,
            chain_type="stuff",
            retriever=self.vectorstore.as_retriever()
        )
        result = qa.invoke({"query": question})
        return result["result"]

# 使用
kb = LocalKnowledgeBase(
    api_key="your-key",
    docs_path="./knowledge.txt"
)

kb.build_index()
answer = kb.query("什么是RAG？")
print(answer)
```
