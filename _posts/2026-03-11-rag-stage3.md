## 1. Embedding 向量嵌入

### 什么是 Embedding？

Embedding 将文本转换为向量表示，让语义相似的文本在向量空间中相近。

```python
from sentence_transformers import SentenceTransformer

# 中文效果好的模型
model = SentenceTransformer('bge-base-zh-v1.5')

# 生成向量
texts = ["Python是一门编程语言", "JavaScript是网页脚本语言"]
embeddings = model.encode(texts)

print(f"向量维度: {embeddings.shape}")
print(f"向量示例: {embeddings[0][:5]}")
```

### 常用 Embedding 模型

| 模型 | 维度 | 特点 | 适用场景 |
|------|------|------|----------|
| text-embedding-ada-002 | 1536 | OpenAI 官方 | 通用 |
| bge-base-zh-v1.5 | 768 | 中文效果优秀 | 中文 |
| m3e-base | 768 | 多语言支持 | 多语言 |
| bge-large-zh-v1.5 | 1024 | 效果最好 | 精度要求高 |

### Chunk 策略

```python
# 1. 固定大小分块
def fixed_chunk(text: str, chunk_size: int = 500, overlap: int = 50):
    chunks = []
    for i in range(0, len(text), chunk_size - overlap):
        chunks.append(text[i:i + chunk_size])
    return chunks

# 2. 递归分块
import re

def recursive_chunk(text: str, delimiters=["\n\n", "\n", ". ", " "]):
    if not delimiters:
        return [text]

    delimiter = delimiters[0]
    parts = text.split(delimiter)

    if len(parts) > 1:
        result = []
        for part in parts:
            result.extend(recursive_chunk(part, delimiters[1:]))
        return result
    else:
        return recursive_chunk(text, delimiters[1:])

# 3. 按段落分块
def paragraph_chunk(text: str):
    paragraphs = text.split("\n\n")
    return [p.strip() for p in paragraphs if p.strip()]
```

### Chroma（轻量级）

```python
import chromadb

# 初始化

collection = client.create_collection("docs")

    "Python是一门易学的编程语言",
    "JavaScript主要用于网页开发",
    "Rust是一门系统编程语言"

    documents=documents,
    embeddings=embeddings.tolist(),
    ids=["doc1", "doc2", "doc3"]

# 检索

query_embedding = model.encode([query])

results = collection.query(
    query_embeddings=query_embedding.tolist(),

```

### Qdrant（云原生）

```python
from qdrant_client import QdrantClient
from qdrant_client.models import Distance, VectorParams
from sentence_transformers import SentenceTransformer

# 连接 Qdrant
client = QdrantClient(host="localhost", port=6333)

# 创建集合

    collection_name="docs",
    vectors_config=VectorParams(size=768, distance=Distance.COSINE)
)

documents = ["文档内容1", "文档内容2"]

client.upsert(
    collection_name="docs",
    points=[
        {
            "id": 1,
            "vector": embeddings[0].tolist(),
            "payload": {"text": documents[0]}
        },
        {
            "id": 2,
            "vector": embeddings[1].tolist(),
            "payload": {"text": documents[1]}
        }
    ]
)

# 检索
query = "用户问题"
query_vector = model.encode([query])[0].tolist()

results = client.search(
    collection_name="docs",
    query_vector=query_vector,
    limit=3
)
```

### Rerank (重排序)

```python
from langchain_community.cross_encoders import HuggingFaceCrossEncoder
from langchain.retrievers import ContextualCompressionRetriever
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings

# 基础检索器
base_retriever = Chroma(
    persist_directory="./chroma_db",
    embedding_function=OpenAIEmbeddings()
).as_retriever()

# Rerank 模型
reranker = HuggingFaceCrossEncoder(model_name="BAAI/bge-reranker-base")

def rerank_documents(query: str, documents: list, top_k: int = 3):
    # 计算相关性分数
    doc_texts = [doc.page_content for doc in documents]
    scores = reranker.rank(query, doc_texts, return_documents=True)

    # 返回重排序后的文档
    return [doc for doc, score in scores[:top_k]]

# 使用
initial_docs = base_retriever.invoke("Python教程")
reranked_docs = rerank_documents("Python教程", initial_docs, top_k=3)
```

## 5. RAG 评估指标

### 核心评估维度

| 维度 | 指标 | 描述 |
|------|------|------|
| 检索质量 | Precision@K | Top-K 结果中相关文档的比例 |
| 检索质量 | Recall@K | 召回的相关文档比例 |
| 检索质量 | MRR | 第一个相关文档的排名倒数 |
| 生成质量 | Faithfulness | 生成内容是否基于检索结果 |
| 生成质量 | Answer Relevance | 回答与问题的相关性 |
| 整体 | RAGAS | 综合评估 RAG 系统 |

### 使用 RAGAs 评估

```python

from datasets import Dataset

# 准备评估数据
```

### 手动评估函数

```python
def evaluate_retrieval(retrieved_docs: list, relevant_docs: list) -> dict:
    """评估检索质量"""
    retrieved_ids = set([doc.metadata.get("id") for doc in retrieved_docs])
    relevant_ids = set([doc.metadata.get("id") for doc in relevant_docs])

    # Precision
    precision = len(retrieved_ids & relevant_ids) / len(retrieved_ids) if retrieved_ids else 0

    # Recall
    recall = len(retrieved_ids & relevant_ids) / len(relevant_ids) if relevant_ids else 0

    # F1
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0

    return {"precision": precision, "recall": recall, "f1": f1}

def evaluate_faithfulness(answer: str, contexts: list) -> float:
    """评估答案是否基于上下文"""
    # 简化实现：检查答案中的关键信息是否在上下文中
    answer_facts = extract_facts(answer)  # 提取答案中的事实
    context_text = " ".join([ctx.page_content for ctx in contexts])

    faithfulness_score = sum(
        1 for fact in answer_facts if fact.lower() in context_text.lower()
    ) / len(answer_facts) if answer_facts else 0

    return faithfulness_score
```

## 6. Agentic RAG

### 什么是 Agentic RAG？

Agentic RAG 将 Agent 能力引入 RAG 系统，实现更智能的检索和问答。

### 自主决定是否检索

```python
from langchain_openai import ChatOpenAI
from langchain.agents import AgentType, create_openai_functions_agent
from langchain.prompts import PromptTemplate
from langchain.tools import Tool
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings

llm = ChatOpenAI(api_key="your-key")
vectorstore = Chroma(persist_directory="./chroma_db", embedding_function=OpenAIEmbeddings())

# 定义检索工具
def retrieve_documents(query: str) -> str:
    docs = vectorstore.similarity_search(query, k=4)
    return "\n\n".join([doc.page_content for doc in docs])

retriever_tool = Tool(
    name="document_retriever",
    func=retrieve_documents,
    description="当需要从知识库检索信息时使用"
)

# 创建 Agent
prompt = PromptTemplate.from_template("""你是一个智能助手。
你有一个文档检索工具可以查询知识库。

用户问题: {input}

你可以：
1. 如果问题需要具体信息，先使用检索工具
2. 如果已经有足够信息，直接回答

{agent_scratchpad}
""")

agent = create_openai_functions_agent(
    llm=llm,
    tools=[retriever_tool],
    prompt=prompt
)

# 执行
result = agent.invoke({"input": "Python中的装饰器是什么？"})
print(result["output"])
```

### 多步推理 RAG

```python
from langchain_openai import ChatOpenAI
from langchain.prompts import PromptTemplate
from langchain.schema import Document
from langgraph.graph import StateGraph, END
from typing import TypedDict

llm = ChatOpenAI(api_key="your-key")

# 状态定义
class RAGState(TypedDict):
    question: str
    sub_questions: list
    retrieved_docs: list
    answer: str

# 分解问题
def decompose_question(state: RAGState):
    question = state["question"]

    # 让 LLM 分解问题
    prompt = f"将这个问题分解成多个子问题:\n{question}"
    response = llm.invoke(prompt)

    sub_questions = response.content.split("\n")
    return {"sub_questions": sub_questions}

# 分别检索
def retrieve_all(state: RAGState):
    docs = []
    for sub_q in state["sub_questions"]:
        sub_docs = vectorstore.similarity_search(sub_q, k=2)
        docs.extend(sub_docs)

    # 去重
    unique_docs = {doc.page_content: doc for doc in docs}.values()
    return {"retrieved_docs": list(unique_docs)}

# 生成答案
def generate_answer(state: RAGState):
    context = "\n\n".join([doc.page_content for doc in state["retrieved_docs"]])

    prompt = f"""基于以下上下文回答问题:
问题: {state['question']}

上下文:
{context}
"""
    response = llm.invoke(prompt)
    return {"answer": response.content}

# 构建图
graph = StateGraph(RAGState)
graph.add_node("decompose", decompose_question)
graph.add_node("retrieve", retrieve_all)
graph.add_node("generate", generate_answer)

graph.set_entry_point("decompose")
graph.add_edge("decompose", "retrieve")
graph.add_edge("retrieve", "generate")
graph.add_edge("generate", END)

agentic_rag = graph.compile()

# 执行
result = agentic_rag.invoke({
    "question": "Python装饰器和JavaScript装饰器有什么区别?",
    "sub_questions": [],
    "retrieved_docs": [],
    "answer": ""
})

print(result["answer"])
```

### 知识库自动更新

```python
from langchain_openai import ChatOpenAI
from langchain_community.document_loaders import TextLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.vectorstores import Chroma

llm = ChatOpenAI(api_key="your-key")

def check_and_update_kb(new_doc_path: str, vectorstore, threshold: float = 0.9):
    """检查文档是否需要更新知识库"""

    # 读取新文档
    loader = TextLoader(new_doc_path)
    new_docs = loader.load()
    new_text = new_docs[0].page_content

    # 检查与现有知识的相似度
    existing_docs = vectorstore.get()
    existing_texts = [doc["page_content"] for doc in existing_docs["documents"][0]]

    # 使用 LLM 判断是否需要更新
    prompt = f"""判断新文档是否引入了新知识:

新文档摘要: {new_text[:500]}

现有知识库摘要: {existing_texts[:3]}

如果新文档包含现有知识库没有的重要新信息，返回"需要更新"，否则返回"无需更新"。
"""
    response = llm.invoke(prompt)

    if "需要更新" in response.content:
        # 分割并添加新文档
        splitter = RecursiveCharacterTextSplitter(chunk_size=500)
        chunks = splitter.split_documents(new_docs)
        vectorstore.add_documents(chunks)
        return "知识库已更新"

    return "知识库无需更新"
```

### 项目：本地知识库问答

```python
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.document_loaders import TextLoader
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain.chains import RetrievalQA

class LocalKnowledgeBase:
    def __init__(self, api_key: str, docs_path: str):
        self.llm = ChatOpenAI(api_key=api_key)
        self.embeddings = OpenAIEmbeddings(api_key=api_key)
        self.docs_path = docs_path

    def build_index(self):
        """构建知识库索引"""
        loader = TextLoader(self.docs_path)
        documents = loader.load()

        text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=500,
            chunk_overlap=50

        docs = text_splitter.split_documents(documents)

        self.vectorstore = Chroma.from_documents(
            docs,
            self.embeddings,
            persist_directory="./chroma_db"
        )

    def query(self, question: str) -> str:
        """问答"""
        qa = RetrievalQA.from_chain_type(
            llm=self.llm,
            chain_type="stuff",
            retriever=self.vectorstore.as_retriever()
        )
        result = qa.invoke({"query": question})
        return result["result"]

# 使用
kb = LocalKnowledgeBase(
    api_key="your-key",
    docs_path="./knowledge.txt"
)

kb.build_index()
answer = kb.query("什么是RAG？")
print(answer)
```
