## 1. OpenAI API

### 快速开始

```python
import os
from openai import OpenAI

# 方式1: 环境变量
# export OPENAI_API_KEY="your-api-key"
client = OpenAI()

# 方式2: 直接传入
client = OpenAI(api_key="your-api-key")

# 基础对话
response = client.chat.completions.create(
    model="gpt-4o",
    messages=[
        {"role": "system", "content": "你是一个有帮助的助手"},
        {"role": "user", "content": "用一句话介绍Python"}
    ]
)

print(response.choices[0].message.content)
```

### 流式输出

```python
from openai import OpenAI

client = OpenAI()

stream = client.chat.completions.create(
    model="gpt-4o",
    messages=[{"role": "user", "content": "写一首关于春天的诗"}],
    stream=True
)

for chunk in stream:
    if chunk.choices[0].delta.content:
        print(chunk.choices[0].delta.content, end="")
```

| 模型 | 特点 | 适用场景 |
|------|------|----------|
| gpt-4o | 最新旗舰多模态 | 复杂任务 |
| gpt-4-turbo | 高性价比 | 日常开发 |
| gpt-3.5-turbo | 快速、便宜 | 简单对话 |

## 2. Anthropic API (Claude)

### 快速开始

```python
from anthropic import Anthropic

client = Anthropic(api_key="your-api-key")

message = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    messages=[
        {"role": "user", "content": "用一句话介绍Python"}
    ]
)

print(message.content[0].text)
```

### 流式输出

```python
from anthropic import Anthropic

client = Anthropic()

with client.messages.stream(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    messages=[{"role": "user", "content": "解释什么是RAG"}]
) as stream:
    for text in stream.text_stream:
        print(text, end="")
```

### Claude 模型选择

| 模型 | 能力 | 速度 | 推荐场景 |
|------|------|------|----------|
| claude-opus-4-6 | 最强 | 慢 | 复杂推理 |
| claude-sonnet-4-6 | 均衡 | 中 | 日常开发 |
| claude-haiku-4-5 | 基础 | 快 | 简单任务 |

### 结构化输出

```python

    name: str
    ingredients: list[str]
    steps: list[str]

    model="gpt-4o",
    messages=[{"role": "user", "content": "教我做蛋炒饭"}],

```

### Prompt 最佳实践

1. **明确任务**：告诉 AI 要做什么
2. **提供上下文**：背景信息有助于理解
3. **使用分隔符**：用 ``` 或 === 分隔不同部分
4. **指定格式**：告诉 AI 输出格式
5. **分步思考**：复杂问题拆成小步骤

## 5. 参数掌握

### 核心参数

```python
response = client.chat.completions.create(
    model="gpt-4o",
    messages=messages,

    # temperature: 0-2（OpenAI），部分模型 0-1（Anthropic），越高越随机
    temperature=0.7,

    # top_p: 0-1，控制采样多样性
    # 与 temperature 二选一使用
    top_p=0.9,

    # max_tokens: 最大输出 token 数
    max_tokens=2048,

    # stop: 遇到这些词停止生成
    stop=["END", "###"],

    # presence_penalty: 惩罚重复内容
    presence_penalty=0.0,

    # frequency_penalty: 惩罚高频词
    frequency_penalty=0.0,
)
```

### 参数选择指南

| 参数 | 低值 (0-0.3) | 中值 (0.4-0.7) | 高值 (0.8-2.0) | 说明 |
|------|-------------|----------------|----------------|------|
| **temperature** | 确定性问题 | 日常对话 | 创意写作 | 控制输出随机性 |
| **top_p** | 保守输出 | 平衡 | 创意输出 | 核采样控制多样性 |

> ⚠️ 注意：Anthropic (Claude) 的 temperature 范围是 **0-1**，不是 0-2

### 常见场景参数

```python
# 代码生成 - 精确
{
    "temperature": 0.2,
    "top_p": 0.95,
}

# 创意写作 - 灵活
{
    "temperature": 0.9,
    "top_p": 0.95,
}

# 摘要总结 - 平衡
{
    "temperature": 0.5,
    "max_tokens": 500,
}
```

## 10. 实战项目

### 项目1：AI 对话助手

```python
from openai import OpenAI

class ChatAssistant:
    def __init__(self, api_key: str, system_prompt: str = "你是一个有帮助的助手"):
        self.client = OpenAI(api_key=api_key)
        self.system_prompt = system_prompt
        self.history = [{"role": "system", "content": system_prompt}]

    def chat(self, message: str) -> str:
        self.history.append({"role": "user", "content": message})

        response = self.client.chat.completions.create(
            model="gpt-4o",
            messages=self.history,
            temperature=0.7
        )

        reply = response.choices[0].message.content
        self.history.append({"role": "assistant", "content": reply})
        return reply

    def clear_history(self):
        self.history = [{"role": "system", "content": self.system_prompt}]

# 使用
assistant = ChatAssistant(
    api_key="your-api-key",
    system_prompt="你是一个Python专家，擅长解释代码"
)

print(assistant.chat("什么是装饰器？"))
```

```python
import os
```

## 7. Function Calling (函数调用)

### 什么是 Function Calling？

Function Calling 允许 LLM 调用外部工具和函数，实现与外部系统的集成。

### OpenAI Function Calling

```python
from openai import OpenAI
from pydantic import BaseModel

client = OpenAI(api_key="your-key")

# 定义函数
tools = [
    {
        "type": "function",
        "function": {
            "name": "get_weather",
            "description": "获取指定城市的天气信息",
            "parameters": {
                "type": "object",
                "properties": {
                    "city": {
                        "type": "string",
                        "description": "城市名称，如北京、上海"
                    }
                },
                "required": ["city"]
            }
        }
    },
    {
        "type": "function",
        "function": {
            "name": "get_stock_price",
            "description": "获取股票价格",
            "parameters": {
                "type": "object",
                "properties": {
                    "symbol": {
                        "type": "string",
                        "description": "股票代码，如 AAPL、GOOGL"
                    }
                },
                "required": ["symbol"]
            }
        }
    }
]

# 调用
response = client.chat.completions.create(
    model="gpt-4o",
    messages=[{"role": "user", "content": "北京天气怎么样？"}],
    tools=tools
)

# 解析函数调用
tool_call = response.choices[0].message.tool_calls[0]
print(f"调用函数: {tool_call.function.name}")
print(f"参数: {tool_call.function.arguments}")

# 模拟执行函数
def get_weather(city: str) -> dict:
    return {"city": city, "weather": "晴", "temperature": 25}

# 执行并返回结果
function_result = eval(f"{tool_call.function.name}({tool_call.function.arguments})")

# 再次调用，获取最终回复
final_response = client.chat.completions.create(
    model="gpt-4o",
    messages=[
        {"role": "user", "content": "北京天气怎么样？"},
        response.choices[0].message,
        {
            "role": "tool",
            "tool_call_id": tool_call.id,
            "content": str(function_result)
        }
    ]
)

print(final_response.choices[0].message.content)
```

### Pydantic 方式（推荐）

```python
from pydantic import BaseModel
from openai import OpenAI

client = OpenAI(api_key="your-key")

# 定义 Pydantic 模型
class WeatherInput(BaseModel):
    city: str

class GetWeather(BaseModel):
    name: str = "get_weather"
    description: str = "获取指定城市的天气信息"
    parameters: type[WeatherInput] = WeatherInput

# 使用 beta API
response = client.beta.chat.completions.parse(
    model="gpt-4o",
    messages=[{"role": "user", "content": "上海明天天气如何？"}],
    tools=[GetWeather]
)

# 获取函数调用
tool_calls = response.choices[0].message.tool_calls
if tool_calls:
    args = tool_calls[0].function.parsed_arguments
    print(f"城市: {args.city}")
```


## 8. Token 计算与成本优化

### Token 计算

```python
import tiktoken

# OpenAI 模型使用 cl100k_base
encoding = tiktoken.get_encoding("cl100k_base")

def count_tokens(text: str) -> int:
    return len(encoding.encode(text))

def count_tokens_messages(messages: list, model: str = "gpt-4") -> int:
    """计算消息列表的 token 数量"""
    if "gpt-4" in model:
        tokens_per_message = 3  # 每条消息的 overhead
        tokens_per_name = 1
    else:
        tokens_per_message = 4
        tokens_per_name = -1

    num_tokens = 0
    for message in messages:
        num_tokens += tokens_per_message
        for key, value in message.items():
            num_tokens += count_tokens(value)
            if key == "name":
                num_tokens += tokens_per_name

    # 添加 completion 的 buffer
    num_tokens += 2
    return num_tokens

# 示例
messages = [
    {"role": "system", "content": "你是一个有帮助的助手"},
    {"role": "user", "content": "用一句话介绍Python"}
]

print(f"Token 数量: {count_tokens_messages(messages)}")
```

### 成本计算

```python
# OpenAI 价格（单位：每百万 token）
PRICING = {
    "gpt-4o": {"input": 5.0, "output": 15.0},
    "gpt-4-turbo": {"input": 10.0, "output": 30.0},
    "gpt-3.5-turbo": {"input": 0.5, "output": 1.5},
}

def calculate_cost(
    model: str,
    input_tokens: int,
    output_tokens: int
) -> float:
    """计算 API 调用成本（美元）"""
    pricing = PRICING.get(model, PRICING["gpt-3.5-turbo"])
    input_cost = (input_tokens / 1_000_000) * pricing["input"]
    output_cost = (output_tokens / 1_000_000) * pricing["output"]
    return input_cost + output_cost

# 示例
input_tokens = 1000  # 1000 tokens
output_tokens = 500   # 500 tokens
cost = calculate_cost("gpt-4o", input_tokens, output_tokens)
print(f"成本: ${cost:.4f}")
```

### 成本优化策略

```python
# 1. 使用合适的模型
model_mapping = {
    "simple": "gpt-3.5-turbo",  # 简单问题
    "normal": "gpt-4-turbo",     # 一般问题
    "complex": "gpt-4o",         # 复杂问题
}

def select_model(task_complexity: str) -> str:
    return model_mapping.get(task_complexity, "gpt-3.5-turbo")

# 2. 压缩上下文
def compress_messages(messages: list, max_tokens: int = 4000):
    """压缩消息历史，保留最新内容"""
    while count_tokens_messages(messages) > max_tokens:
        if len(messages) <= 2:  # 保留 system 和最后一条
            break
        messages.pop(1)  # 删除最旧的用户消息
    return messages

# 3. 使用缓存
def cached_completion(cache: dict, client, model: str, messages: list):
    """简单缓存实现"""
    cache_key = f"{model}:{messages[-1]['content'][:50]}"
    if cache_key in cache:
        return cache[cache_key]

    response = client.chat.completions.create(model=model, messages=messages)
    cache[cache_key] = response
    return response
```
